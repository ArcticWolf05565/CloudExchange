class StringUtils{
    public:
    std::string format(const std::string& text){
        std::string temp = text;
        std::transform(temp.begin() , temp.end() , temp.begin(),::toupper);
       return temp;
    }
    std::string format(const std::string& text, int repeateCount){

        if(repeateCount <= 0) return "";
        std::string result = text;

        while(repeateCount--){
          result += " " + text;
        }
        return result;
    }
    std::string format(const std::string& text, bool addBrackters){
        if(addBrackters){
            return ("[" + text + "]");
        }else{   
            return text;
        }
    }
    std::string format(const std::string& text, const std::string& prefix, const std::string& suffix){
        return (prefix + text  + suffix);
    }
    ~StringUtils() = default;
};

    std::cout<< test.format("rahul", true)<<std::endl;

    return 0;

    void removeSong(std::string songname){

            }

    }

        std::string allSongs;

            allSongs += songName[i];

            if(i < songName.size() -1){
                allSongs += "," ;
            }
        }
        return allSongs;
    }
    
};

class BrowserHistory{
    private:
    std::string browserName;
    std::stack<std::string> history;

    public:
    BrowserHistory(std::string browserName) : browserName(browserName){}

    void visitPage(std::string url){
        history.push(url);
    }

    std::string goBack(){
        std::string url;
        if(!history.empty()){
            url = history.top();
            history.pop();
            return url;
        }else{
            return "No history found.\n";
        }
    }

    std::string getBrowserName(){
        return browserName;
    }
    int getHistorySize(){
        return history.size();
    }

};

int main(){

    BrowserHistory surfing("Chrome");

    std::cout<< "Browser name: " << surfing.getBrowserName() << "\n";

    surfing.visitPage("Flipkart.in");

    SupportTicket(std::string ticketID, std::string customerName, std::string issueDescription) : ticketID(ticketID) , customerName(customerName) , issueDescription(issueDescription){}

    std::string getTicketID(){
        return ticketID;
    }
    std::string getCustomerName(){
        return customerName;
    }
    std::string getIssueDescription(){
        return issueDescription;
    }

    virtual std::string getPriority() = 0;

    std::string getTicketSummary(){
        return ("Ticket: " + ticketID + " Customer name: " + customerName + " Issue Description: "  + issueDescription + getPriority());
    }
    virtual ~SupportTicket() = default; 
};
class UrgentTicket : public SupportTicket{
    public:

    UrgentTicket(std::string ticketID, std::string customerName, std::string issueDescription) : SupportTicket(ticketID, customerName, issueDescription){}

    std::string getPriority()override{
        return " | Priority: HIGH";
    }
};
class NormalTicket : public SupportTicket{
    public:
    std::string getPriority()override{
        return " | Priority: LOW";
    }
     NormalTicket(std::string ticketID, std::string customerName, std::string issueDescription) : SupportTicket(ticketID, customerName, issueDescription){}
};
class TicketQueue{
    private:
    std::queue<SupportTicket*> tkt;

    public:

    void addTicket(SupportTicket* ticket){
        tkt.push(ticket);
    }
    int getQueueSize(){
        return tkt.size();
    }
    SupportTicket* processNextTicket(){
        SupportTicket* ptrTkt = nullptr;
        if(!tkt.empty()){
            ptrTkt = tkt.front();
            tkt.pop();
            return ptrTkt;
        }else{
            return ptrTkt;
        }
  
    }

};

int main(){

  TicketQueue qtkt;

  UrgentTicket* utkt = new UrgentTicket("A23CG89", "Furkan", "Fazil");

  qtkt.addTicket(ntkt);
