    std::string format(const std::string& text){
        std::string temp = text;
        std::transform(temp.begin() , temp.end() , temp.begin(),::toupper);
       return temp;
    }
    std::string format(const std::string& text, int repeateCount){

        if(repeateCount <= 0) return "";
        std::string result = text;

        while(repeateCount--){
          result += " " + text;
        }
        return result;
    }
    std::string format(const std::string& text, bool addBrackters){
        if(addBrackters){
            return ("[" + text + "]");
        }else{   
            return text;
        }
    }
    std::string format(const std::string& text, const std::string& prefix, const std::string& suffix){
        return (prefix + text  + suffix);
    }
    ~StringUtils() = default;
};

    std::string browserName;
    std::stack<std::string> history;

    public:
    BrowserHistory(std::string browserName) : browserName(browserName){}

    void visitPage(std::string url){
        history.push(url);
    }

    std::string goBack(){
        std::string url;
        if(!history.empty()){
            url = history.top();
            history.pop();
            return url;
        }else{
            return "No history found.\n";
        }
    }

    std::string getBrowserName(){
        return browserName;
    }
    int getHistorySize(){
        return history.size();
    }

};

    }

    }
    virtual ~SupportTicket() = default; 
};
class UrgentTicket : public SupportTicket{
    public:

    UrgentTicket(std::string ticketID, std::string customerName, std::string issueDescription) : SupportTicket(ticketID, customerName, issueDescription){}

    std::string getPriority()override{
        return " | Priority: HIGH";
    }
};
class NormalTicket : public SupportTicket{
    public:
    std::string getPriority()override{
        return " | Priority: LOW";
    }
     NormalTicket(std::string ticketID, std::string customerName, std::string issueDescription) : SupportTicket(ticketID, customerName, issueDescription){}
};
class TicketQueue{
    private:
    std::queue<SupportTicket*> tkt;

    public:

    void addTicket(SupportTicket* ticket){
        tkt.push(ticket);
    }
    int getQueueSize(){
        return tkt.size();
    }
    SupportTicket* processNextTicket(){
        SupportTicket* ptrTkt = nullptr;
        if(!tkt.empty()){
            ptrTkt = tkt.front();
            tkt.pop();
            return ptrTkt;
        }else{
            return ptrTkt;
        }
  
    }

};

#include <iostream>
#include <string>
#include <sstream>
#include <stack>

class CodeEditor{
    private:
    std::string filename;
    std::ostringstream currentCodeContent;
    std::stack<std::string> preStateSnapshots;

    public:
    CodeEditor(std::string filename) : filename(filename){
        currentCodeContent.str("");
    }
    void typeCode(std::string code){
        preStateSnapshots.push(currentCodeContent.str());

        currentCodeContent << code << "\n";
    }
    std::string undo(){
        std::string lastSaved;
        if(preStateSnapshots.empty()){
            return "Nothing to undo";
        }

        lastSaved = preStateSnapshots.top();
        preStateSnapshots.pop();
        currentCodeContent.str("");
        currentCodeContent.clear();

        currentCodeContent << lastSaved;
        return "undo successful";
    }
    std::string getCurrentCode(){return currentCodeContent.str();}

    std::string getFilename(){return filename;}

};

int main(){

    CodeEditor c1("Notepad");

    c1.typeCode("int x = 5;");
    c1.typeCode("int *ptr = &x");
    c1.typeCode("std::cout<< ptr << std::endl;");

    std::cout << "\n" << c1.getCurrentCode() << std::endl;

    c1.undo();
    std::cout << "\n" << c1.getCurrentCode() << std::endl;
    c1.undo();
    std::cout << "\n" << c1.getCurrentCode() << std::endl;

    return 0;
}
