class Recordable{
    public:

    virtual std::string startRecording() = 0;
    virtual std::string stopRecording() = 0;
    virtual ~Recordable() = default;
};
class MotionDetectable{
    public:
    virtual std::string detectMotion(const std::string& zone) = 0;
    virtual ~MotionDetectable() = default;
};
class Alertable{
    public:
    virtual std::string triggerAlert(const std::string& reason) = 0;
    virtual ~Alertable() = default;
};
class SecurityCamera : public Recordable, public MotionDetectable , public Alertable{
    private:
    std::string cameraId;

    public:
    SecurityCamera(std::string cameraId) : cameraId(cameraId){}
    std::string startRecording()override{return (cameraId + ": Recording Start..."); }
    std::string stopRecording()override{return (cameraId + ": Recording Stop...");}
    std::string detectMotion(const std::string& zone)override{return (cameraId + " " + zone + ": Motion Detected.");}
    std::string triggerAlert(const std::string& reason)override{return (cameraId + ": " + reason);}
};

int main(){

    SecurityCamera cam("2341");

    std::cout<< cam.startRecording() <<std::endl;

    std::cout<< cam.triggerAlert("Suspicious Activity")<< std::endl;
    std::cout<< cam.stopRecording()<< std::endl;

    return 0;

#include <iostream>
#include <string>
#include <algorithm>
#include <cctype>

class StringUtils{
    public:
    std::string format(const std::string& text){
        std::string temp = text;
        std::transform(temp.begin() , temp.end() , temp.begin(),::toupper);
       return temp;
    }
    std::string format(const std::string& text, int repeateCount){

        if(repeateCount <= 0) return "";
        std::string result = text;

        while(repeateCount--){
          result += " " + text;
        }
        return result;
    }
    std::string format(const std::string& text, bool addBrackters){
        if(addBrackters){
            return ("[" + text + "]");
        }else{   
            return text;
        }
    }
    std::string format(const std::string& text, const std::string& prefix, const std::string& suffix){
        return (prefix + text  + suffix);
    }
    ~StringUtils() = default;
};

int main(){

    StringUtils test;

    std::cout<< test.format("hello")<< std::endl;
    std::cout << test.format("kali" , 3) << std::endl;
    std::cout<< test.format("rahul", true)<<std::endl;
    std::cout<< test.format("2", "H" , "O");
    return 0;
}

class Teacher{

    std::string teacherName;
    std::string subject;

    public:
    Teacher(std::string teacherName, std::string subject) : teacherName(teacherName) , subject(subject){};

    std::pair<std::string , std::string> getTeacherInfo() const{
        return {teacherName, subject};
    }
};
class Classroom{
    private:
    std::string roomNumber;
    int capacity;
    Teacher faculty;

    public:
    Classroom(std::string roomNumber, int capacity, Teacher faculty): roomNumber(roomNumber), capacity(capacity) , faculty(faculty){};

    std::string getClassroomInfo() const{
     const auto [tname, sub] = faculty.getTeacherInfo();
        return ("Teacher name: " + tname + "\n"
                + "Subject: " + sub + "\n"
                + "Room no: " + roomNumber + "\n"
                + "Capacity: " + std::to_string(capacity));
    }
};
class School{
    private:
    std::string schoolName;
    std::vector<Classroom> classroom;

    public:
    School(std::string schoolName): schoolName(schoolName){};

    void addClassroom(Classroom classroom){
        this->classroom.push_back(classroom);
    }

    std::string getSchoolReport(){
        std::string report = "School name: " + schoolName + "\n---------------------------------\n";

        for(const auto& room: classroom){
            report += room.getClassroomInfo() + "\n---------------------------------\n";
        }
        return report;
    }
};

    Teacher sir("Sartaj", "Maths");
    Teacher mam("Arshuma", "English");

    Classroom young("A1", 50, sir);
    Classroom older("B2", 90, mam);

    School edu("Arshi public school");
    edu.addClassroom(young);
    edu.addClassroom(older);

class Playlist{
    private:
    std::string playlist;
    std::vector<std::string> songName;

    public:
    Playlist(std::string playlist) : playlist(playlist){}

    void addSong(std::string songname){
        songName.push_back(songname);
    }
    void removeSong(std::string songname){
        for(auto it = songName.begin() ; it != songName.end() ; ++it){
            if(*it == songname){
                songName.erase(it);
                break;
            }
        }
    }
    std::string getPlaylistName(){
        return playlist;
    }
    std::string getAllSongs(){
       
        std::string allSongs;
        for(size_t i = 0; i < songName.size() ; ++i){
            allSongs += songName[i];

            if(i < songName.size() -1){
                allSongs += "," ;
            }
        }
        return allSongs;
    }
    
};

int main(){

    Playlist music("My Playlist");

    music.addSong("Faded");
    music.addSong("You belong with me");
    music.addSong("sapphire");

    music.removeSong("Faded");

    std::cout << music.getAllSongs() << std::endl;
    return 0;

#include <iostream>
#include <string>
#include <stack>

class BrowserHistory{
    private:
    std::string browserName;
    std::stack<std::string> history;

    public:
    BrowserHistory(std::string browserName) : browserName(browserName){}

    void visitPage(std::string url){
        history.push(url);
    }

    std::string goBack(){
        std::string url;
        if(!history.empty()){
            url = history.top();
            history.pop();
            return url;
        }else{
            return "No history found.\n";
        }
    }

    std::string getBrowserName(){
        return browserName;
    }
    int getHistorySize(){
        return history.size();
    }

};

int main(){

    BrowserHistory surfing("Chrome");

    std::cout<< "Browser name: " << surfing.getBrowserName() << "\n";
    surfing.visitPage("google.com");
    surfing.visitPage("Flipkart.in");
    surfing.visitPage("moi.org");

    std::cout<< "Browser history size: " << surfing.getHistorySize() << "\n";
    std::cout<< surfing.goBack() << "\n";
    std::cout<< "Browser history size: " << surfing.getHistorySize() << "\n";
    std::cout<< surfing.goBack() << "\n";
    std::cout<< "Browser history size: " << surfing.getHistorySize() << "\n";

    return 0;
}
